047/**
048 * <p>Simple implementation of {@link Logger} that sends all enabled log messages,
049 * for all defined loggers, to the console ({@code System.err}).
050 * The following system properties are supported to configure the behavior of this logger:</p>
051 *
052 * <ul>
053 * <li><code>org.slf4j.simpleLogger.logFile</code> - The output target which can be the <em>path</em> to a file, or
054 * the special values "System.out" and "System.err". Default is "System.err".
055 *
056 * <li><code>org.slf4j.simpleLogger.defaultLogLevel</code> - Default log level for all instances of SimpleLogger.
057 * Must be one of ("trace", "debug", "info", "warn", "error" or "off"). If not specified, defaults to "info". </li>
058 *
059 * <li><code>org.slf4j.simpleLogger.log.<em>a.b.c</em></code> - Logging detail level for a SimpleLogger instance
060 * named "a.b.c". Right-side value must be one of "trace", "debug", "info", "warn", "error" or "off". When a SimpleLogger
061 * named "a.b.c" is initialized, its level is assigned from this property. If unspecified, the level of nearest parent
062 * logger will be used, and if none is set, then the value specified by
063 * <code>org.slf4j.simpleLogger.defaultLogLevel</code> will be used.</li>
064 *
065 * <li><code>org.slf4j.simpleLogger.showDateTime</code> - Set to <code>true</code> if you want the current date and
066 * time to be included in output messages. Default is <code>false</code></li>
067 *
068 * <li><code>org.slf4j.simpleLogger.dateTimeFormat</code> - The date and time format to be used in the output messages.
069 * The pattern describing the date and time format is defined by
070 * <a href="http://docs.oracle.com/javase/1.5.0/docs/api/java/text/SimpleDateFormat.html"><code>SimpleDateFormat</code></a>.
071 * If the format is not specified or is invalid, the number of milliseconds since start up will be output. </li>
072 *
073 * <li><code>org.slf4j.simpleLogger.showThreadName</code> -Set to <code>true</code> if you want to output the current
074 * thread name. Defaults to <code>true</code>.</li>
075 *
076 * <li><code>org.slf4j.simpleLogger.showLogName</code> - Set to <code>true</code> if you want the Logger instance name
077 * to be included in output messages. Defaults to <code>true</code>.</li>
078 *
079 * <li><code>org.slf4j.simpleLogger.showShortLogName</code> - Set to <code>true</code> if you want the last component
080 * of the name to be included in output messages. Defaults to <code>false</code>.</li>
081 *
082 * <li><code>org.slf4j.simpleLogger.levelInBrackets</code> - Should the level string be output in brackets? Defaults
083 * to <code>false</code>.</li>
084 *
085 * <li><code>org.slf4j.simpleLogger.warnLevelString</code> - The string value output for the warn level. Defaults
086 * to <code>WARN</code>.</li>
087
088 * </ul>
089 *
090 * <p>In addition to looking for system properties with the names specified above, this implementation also checks for
091 * a class loader resource named <code>"simplelogger.properties"</code>, and includes any matching definitions
092 * from this resource (if it exists).</p>
093 *
094 * <p>With no configuration, the default output includes the relative time in milliseconds, thread name, the level,
095 * logger name, and the message followed by the line separator for the host.  In log4j terms it amounts to the "%r [%t]
096 * %level %logger - %m%n" pattern. </p>
097 * <p>Sample output follows.</p>
098 * <pre>
099 * 176 [main] INFO examples.Sort - Populating an array of 2 elements in reverse order.
100 * 225 [main] INFO examples.SortAlgo - Entered the sort method.
101 * 304 [main] INFO examples.SortAlgo - Dump of integer array:
102 * 317 [main] INFO examples.SortAlgo - Element [0] = 0
103 * 331 [main] INFO examples.SortAlgo - Element [1] = 1
104 * 343 [main] INFO examples.Sort - The next log statement should be an error message.
105 * 346 [main] ERROR examples.SortAlgo - Tried to dump an uninitialized array.
106 *   at org.log4j.examples.SortAlgo.dump(SortAlgo.java:58)
107 *   at org.log4j.examples.Sort.main(Sort.java:64)
108 * 467 [main] INFO  examples.Sort - Exiting main method.
109 * </pre>
110 *
111 * <p>This implementation is heavily inspired by
112 * <a href="http://commons.apache.org/logging/">Apache Commons Logging</a>'s SimpleLog.</p>
113 *
114 * @author Ceki G&uuml;lc&uuml;
115 * @author <a href="mailto:sanders@apache.org">Scott Sanders</a>
116 * @author Rod Waldhoff
117 * @author Robert Burrell Donkin
118 * @author C&eacute;drik LIME
119 */

166    private static String getStringProperty(String name) {
167        String prop = null;
168        try {
169            prop = System.getProperty(name);
170        } catch (SecurityException e) {
171            ; // Ignore
172        }
173        return (prop == null) ? SIMPLE_LOGGER_PROPS.getProperty(name) : prop;
174    }
175
176    private static String getStringProperty(String name, String defaultValue) {
177        String prop = getStringProperty(name);
178        return (prop == null) ? defaultValue : prop;
179    }
180
181    private static boolean getBooleanProperty(String name, boolean defaultValue) {
182        String prop = getStringProperty(name);
183        return (prop == null) ? defaultValue : "true".equalsIgnoreCase(prop);
184    }

194    static void init() {
195        loadProperties();
196
197        String defaultLogLevelString = getStringProperty(DEFAULT_LOG_LEVEL_KEY, null);
198        if (defaultLogLevelString != null)
199            DEFAULT_LOG_LEVEL = stringToLevel(defaultLogLevelString);
200
201        SHOW_LOG_NAME = getBooleanProperty(SHOW_LOG_NAME_KEY, SHOW_LOG_NAME);
202        SHOW_SHORT_LOG_NAME = getBooleanProperty(SHOW_SHORT_LOG_NAME_KEY, SHOW_SHORT_LOG_NAME);
203        SHOW_DATE_TIME = getBooleanProperty(SHOW_DATE_TIME_KEY, SHOW_DATE_TIME);
204        SHOW_THREAD_NAME = getBooleanProperty(SHOW_THREAD_NAME_KEY, SHOW_THREAD_NAME);
205        DATE_TIME_FORMAT_STR = getStringProperty(DATE_TIME_FORMAT_KEY, DATE_TIME_FORMAT_STR);
206        LEVEL_IN_BRACKETS = getBooleanProperty(LEVEL_IN_BRACKETS_KEY, LEVEL_IN_BRACKETS);
207        WARN_LEVEL_STRING = getStringProperty(WARN_LEVEL_STRING_KEY, WARN_LEVEL_STRING);
208
209        LOG_FILE = getStringProperty(LOG_FILE_KEY, LOG_FILE);
210        OUTPUT_CHOICE = computeOutputChoice(LOG_FILE);
211
212        if (DATE_TIME_FORMAT_STR != null) {
213            try {
214                DATE_FORMATTER = new SimpleDateFormat(DATE_TIME_FORMAT_STR);
215            } catch (IllegalArgumentException e) {
216                Util.report("Bad date format in " + CONFIGURATION_FILE + "; will output relative time", e);
217            }
218        }
219    }

225    private static OutputChoice computeOutputChoice(String logFile) {
226        if ("System.err".equalsIgnoreCase(logFile))
227            return new OutputChoice(OutputChoiceType.SYS_ERR);
228        else if ("System.out".equalsIgnoreCase(logFile)) {
229            return  new OutputChoice(OutputChoiceType.SYS_OUT);
230        } else {
231            try {
232                FileOutputStream fos = new FileOutputStream(logFile);
233                PrintStream printStream = new PrintStream(fos);
234                return new OutputChoice(printStream);
235            } catch (FileNotFoundException e) {
236                Util.report("Could not open [" + logFile + "]. Defaulting to System.err", e);
237                return new OutputChoice(OutputChoiceType.SYS_ERR);
238            }
239        }
240    }

242    private static void loadProperties() {
243        // Add props from the resource simplelogger.properties
244        InputStream in = AccessController.doPrivileged(new PrivilegedAction<InputStream>() {
245            public InputStream run() {
246                ClassLoader threadCL = Thread.currentThread().getContextClassLoader();
247                if (threadCL != null) {
248                    return threadCL.getResourceAsStream(CONFIGURATION_FILE);
249                } else {
250                    return ClassLoader.getSystemResourceAsStream(CONFIGURATION_FILE);
251                }
252            }
253        });
254        if (null != in) {
255            try {
256                SIMPLE_LOGGER_PROPS.load(in);
257            } catch (java.io.IOException e) {
258                // ignored
259            } finally {
260                try {
261                    in.close();
262                } catch (java.io.IOException e) {
263                    // ignored
264                }
265            }
266        }
267    }

274    /**
275     * Package access allows only {@link SimpleLoggerFactory} to instantiate
276     * SimpleLogger instances.
277     */
278    SimpleLogger(String name) {
279        this.name = name;
280
281        String levelString = recursivelyComputeLevelString();
282        if (levelString != null) {
283            this.currentLogLevel = stringToLevel(levelString);
284        } else {
285            this.currentLogLevel = DEFAULT_LOG_LEVEL;
286        }
287    }
288
289    String recursivelyComputeLevelString() {
290        String tempName = name;
291        String levelString = null;
292        int indexOfLastDot = tempName.length();
293        while ((levelString == null) && (indexOfLastDot > -1)) {
294            tempName = tempName.substring(0, indexOfLastDot);
295            levelString = getStringProperty(LOG_KEY_PREFIX + tempName, null);
296            indexOfLastDot = String.valueOf(tempName).lastIndexOf(".");
297        }
298        return levelString;
299    }

301    private static int stringToLevel(String levelStr) {
302        if ("trace".equalsIgnoreCase(levelStr)) {
303            return LOG_LEVEL_TRACE;
304        } else if ("debug".equalsIgnoreCase(levelStr)) {
305            return LOG_LEVEL_DEBUG;
306        } else if ("info".equalsIgnoreCase(levelStr)) {
307            return LOG_LEVEL_INFO;
308        } else if ("warn".equalsIgnoreCase(levelStr)) {
309            return LOG_LEVEL_WARN;
310        } else if ("error".equalsIgnoreCase(levelStr)) {
311            return LOG_LEVEL_ERROR;
312        } else if ("off".equalsIgnoreCase(levelStr)) {
313            return LOG_LEVEL_OFF;
314        }
315        // assume INFO by default
316        return LOG_LEVEL_INFO;
317    }

319    /**
320     * This is our internal implementation for logging regular (non-parameterized)
321     * log messages.
322     *
323     * @param level   One of the LOG_LEVEL_XXX constants defining the log level
324     * @param message The message itself
325     * @param t       The exception whose stack trace should be logged
326     */
327    private void log(int level, String message, Throwable t) {
328        if (!isLevelEnabled(level)) {
329            return;
330        }
331
332        StringBuilder buf = new StringBuilder(32);
333
334        // Append date-time if so configured
335        if (SHOW_DATE_TIME) {
336            if (DATE_FORMATTER != null) {
337                buf.append(getFormattedDate());
338                buf.append(' ');
339            } else {
340                buf.append(System.currentTimeMillis() - START_TIME);
341                buf.append(' ');
342            }
343        }
344
345        // Append current thread name if so configured
346        if (SHOW_THREAD_NAME) {
347            buf.append('[');
348            buf.append(Thread.currentThread().getName());
349            buf.append("] ");
350        }
351
352        if (LEVEL_IN_BRACKETS)
353            buf.append('[');
354
355        // Append a readable representation of the log level
356        switch (level) {
357        case LOG_LEVEL_TRACE:
358            buf.append("TRACE");
359            break;
360        case LOG_LEVEL_DEBUG:
361            buf.append("DEBUG");
362            break;
363        case LOG_LEVEL_INFO:
364            buf.append("INFO");
365            break;
366        case LOG_LEVEL_WARN:
367            buf.append(WARN_LEVEL_STRING);
368            break;
369        case LOG_LEVEL_ERROR:
370            buf.append("ERROR");
371            break;
372        }
373        if (LEVEL_IN_BRACKETS)
374            buf.append(']');
375        buf.append(' ');
376
377        // Append the name of the log instance if so configured
378        if (SHOW_SHORT_LOG_NAME) {
379            if (shortLogName == null)
380                shortLogName = computeShortName();
381            buf.append(String.valueOf(shortLogName)).append(" - ");
382        } else if (SHOW_LOG_NAME) {
383            buf.append(String.valueOf(name)).append(" - ");
384        }
385
386        // Append the message
387        buf.append(message);
388
389        write(buf, t);
390
391    }
392
393    void write(StringBuilder buf, Throwable t) {
394        PrintStream targetStream = OUTPUT_CHOICE.getTargetPrintStream();
395        
396        targetStream.println(buf.toString());
397        if (t != null) {
398            t.printStackTrace(targetStream);
399        }
400        targetStream.flush();
401    }

406        private String getFormattedDate() {
407        Date now = new Date();
408        String dateText;
409        synchronized (DATE_FORMATTER) {
410            dateText = DATE_FORMATTER.format(now);
411        }
412        return dateText;
413    }

415    private String computeShortName() {
416        return name.substring(name.lastIndexOf(".") + 1);
417    }

419    /**
420     * For formatted messages, first substitute arguments and then log.
421     *
422     * @param level
423     * @param format
424     * @param arg1
425     * @param arg2
426     */
427    private void formatAndLog(int level, String format, Object arg1, Object arg2) {
428        if (!isLevelEnabled(level)) {
429            return;
430        }
431        FormattingTuple tp = MessageFormatter.format(format, arg1, arg2);
432        log(level, tp.getMessage(), tp.getThrowable());
433    }
434
435    /**
436     * For formatted messages, first substitute arguments and then log.
437     *
438     * @param level
439     * @param format
440     * @param arguments a list of 3 ore more arguments
441     */
442    private void formatAndLog(int level, String format, Object... arguments) {
443        if (!isLevelEnabled(level)) {
444            return;
445        }
446        FormattingTuple tp = MessageFormatter.arrayFormat(format, arguments);
447        log(level, tp.getMessage(), tp.getThrowable());
448    }

455    protected boolean isLevelEnabled(int logLevel) {
456        // log level are numerically ordered so can use simple numeric
457        // comparison
458        return (logLevel >= currentLogLevel);
459    }

660     * ERROR according to the format outlined above.
661     */
662    public void error(String format, Object... argArray) {
663        formatAndLog(LOG_LEVEL_ERROR, format, argArray);
664    }
665
666    /** Log a message of level ERROR, including an exception. */
667    public void error(String msg, Throwable t) {
668        log(LOG_LEVEL_ERROR, msg, t);
669    }
670
671    public void log(LoggingEvent event) {
672        int levelInt = event.getLevel().toInt();
673
674        if (!isLevelEnabled(levelInt)) {
675            return;
676        }
677        FormattingTuple tp = MessageFormatter.arrayFormat(event.getMessage(), event.getArgumentArray(), event.getThrowable());
678        log(levelInt, tp.getMessage(), event.getThrowable());
679    }
